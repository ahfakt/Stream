178std::uint32_t
179ResolveInet(char const* host)
180{
181 struct addrinfo hints = {0};
182 hints.ai_flags = 0;
183 hints.ai_family = AF_INET;
184 hints.ai_socktype = SOCK_STREAM;
185 hints.ai_protocol = IPPROTO_TCP;
186
187 struct addrinfo* info = nullptr;
188 int e = getaddrinfo(host, nullptr, &hints, &info);
189 if (e == 0) {
190 std::uint32_t r = reinterpret_cast<sockaddr_in*>(info->ai_addr)->sin_addr.s_addr;
191 freeaddrinfo(info);
192 return r;
193 }
194 throw Socket::Inet::Exception(static_cast<Socket::Inet::Exception::Code>(e));
195}
